\section{Methodology}
\label{sec:method}

To address the cross-project generalisation gap and the data-silo problem simultaneously, we propose \emph{VulMorph-Fed}. The framework replaces standard parameter-averaging with a privacy-preserving federated aggregation of semantic structural prototypes. The methodology consists of three core components: (1)~Vulnerability-Critical Subgraph Abstraction (VCSA), which extracts project-invariant structural morphologies; (2)~Morphology-Conditioned Federated Prototype Aggregation (MCFPA), which securely aggregates structural knowledge; and (3)~Morphology-Guided Message Passing (MGMP), which integrates global intelligence into local graph evaluation. Let $\mathcal{G}_k = \{G_i^{(k)}\}_{i=1}^{N_k}$ denote the local dataset of client $k \in \{1, \dots, K\}$, where each graph $G_i^{(k)} = (\mathcal{V}, \mathcal{E}, \mathbf{X})$ represents a function property graph with node features $\mathbf{X} \in \mathbb{R}^{|\mathcal{V}| \times d}$.

\subsection{Vulnerability-Critical Subgraph Abstraction (VCSA)}
Raw Code Property Graphs (CPGs) contain thousands of project-specific API tokens, causing GNNs to overfit to local lexical distributions. VCSA performs a structural abstraction mapping $\Phi: \mathcal{G} \rightarrow \tilde{\mathcal{G}}$ that strips lexical noise while preserving vulnerability-critical data flow. 

We define a finite semantic taxonomy $\mathcal{T} = \{\tau_1, \dots, \tau_8\}$, comprising fundamental operation types (e.g., \texttt{MEMORY\_ALLOC}, \texttt{PTR\_DEREF}, \texttt{ARRAY\_INDEX}). The abstraction function maps each node $v \in \mathcal{V}$ to a semantic type:
\begin{equation}
    \tilde{v} = \phi_{\text{type}}(v) \in \mathcal{T} \cup \{\texttt{UNKNOWN}\}
\end{equation}
We extract the vulnerability-critical subgraph $\tilde{G} \subset G$ by executing a bidirectional reachability query over the data-dependence edges $\mathcal{E}_{DD}$ originating from any node $v$ where $\phi_{\text{type}}(v) \neq \texttt{UNKNOWN}$:
\begin{equation}
    \tilde{\mathcal{V}} = \{ u \in \mathcal{V} \mid \exists \text{ path } u \leftrightarrow v \text{ in } \mathcal{E}_{DD} \}
\end{equation}
The abstracted node features $\tilde{\mathbf{X}} \in \mathbb{R}^{|\tilde{\mathcal{V}}| \times |\mathcal{T}|}$ are initialized as one-hot encodings of their semantic types. The local GNN encoder $f_{\theta_k}: \tilde{\mathcal{G}} \rightarrow \mathbb{R}^{d}$ then computes the graph-level representation:
\begin{equation}
    \mathbf{h}_{\tilde{G}} = \text{READOUT} \left( \text{GNN}_{\theta_k}(\tilde{\mathbf{X}}, \tilde{\mathcal{E}}) \right)
\end{equation}
By operating strictly on $\tilde{G}$, $f_{\theta_k}$ is forced to learn the \emph{shape} of the data flow (the morphology) rather than memorizing project-specific token distributions.

\subsection{Morphology-Conditioned Federated Prototype Aggregation (MCFPA)}
Standard Federated Learning algorithms, such as FedAvg \cite{mcmahan2017fedavg}, aggregate neural network parameters: $\theta_{t+1} = \sum_{k=1}^K \frac{N_k}{N} \theta_t^{(k)}$. This formulation assumes IID data distributions. However, vulnerability datasets exhibit extreme non-IID characteristics, with highly skewed Common Weakness Enumeration (CWE) class distributions across clients.

MCFPA circumvents parameter averaging by federating \emph{semantic prototypes}. Let $\mathcal{C}$ be the global set of CWE classes. For each class $c \in \mathcal{C}$, client $k$ computes a local prototype $\mathbf{p}_c^{(k)} \in \mathbb{R}^d$ as the empirical centroid of the embeddings of functions labeled with $c$:
\begin{equation}
    \mathbf{p}_c^{(k)} = \frac{1}{| \{ G \in \mathcal{G}_k \mid y_G = c \} |} \sum_{G \in \mathcal{G}_k, y_G = c} f_{\theta_k}(\tilde{G})
\end{equation}
To preserve data privacy, we apply the Laplace mechanism. We bound the $L_2$ norm of $\mathbf{h}_{\tilde{G}}$ to $R$, yielding a global sensitivity of $\Delta f = 2R / N_{c,k}$. The perturbed local prototype is:
\begin{equation}
    \hat{\mathbf{p}}_c^{(k)} = \mathbf{p}_c^{(k)} + \text{Lap}\left(0, \frac{\Delta f}{\varepsilon}\right)
\end{equation}
The central server aggregates the perturbed prototypes weighted by the local class frequencies $w_c^{(k)}$, producing the global morphology bank $\mathcal{P}_{global} = \{ \mathbf{p}_c^{global} \}_{c \in \mathcal{C}}$:
\begin{equation}
    \mathbf{p}_c^{global} = \sum_{k=1}^K w_c^{(k)} \hat{\mathbf{p}}_c^{(k)}
\end{equation}

\subsection{Morphology-Guided Message Passing (MGMP)}
During local training and inference, client $k$ retrieves $\mathcal{P}_{global}$ and integrates it into its local GNN via MGMP. Instead of updating local node representations solely based on immediate neighbors $\mathcal{N}(v)$, MGMP computes a global context vector $\mathbf{m}_{global}^{(v)}$ via an attention mechanism over the global prototypes:
\begin{equation}
    \alpha_{v, c} = \frac{\exp \left( \text{LeakyReLU}(\mathbf{a}^T [\mathbf{W} \mathbf{h}_v \Vert \mathbf{p}_c^{global}]) \right)}{\sum_{c' \in \mathcal{C}} \exp \left( \text{LeakyReLU}(\mathbf{a}^T [\mathbf{W} \mathbf{h}_v \Vert \mathbf{p}_{c'}^{global}]) \right)}
\end{equation}
\begin{equation}
    \mathbf{m}_{global}^{(v)} = \sum_{c \in \mathcal{C}} \alpha_{v, c} \mathbf{p}_c^{global}
\end{equation}
The final node representation at layer $l+1$ is then updated by fusing the local neighborhood aggregation with the global context:
\begin{equation}
    \mathbf{h}_v^{(l+1)} = \sigma \left( \mathbf{W}_{local} \sum_{u \in \mathcal{N}(v)} \mathbf{h}_u^{(l)} + \mathbf{W}_{global} \mathbf{m}_{global}^{(v)} \right)
\end{equation}
This mechanism ensures that even if client $k$ has never encountered a specific CWE class locally, its GNN can recognize the structural morphology of that vulnerability by attending to the corresponding global prototype during message passing.

\begin{algorithm}[t]
\caption{VulMorph-Fed Framework}
\label{alg:vulmorph_fed}
\begin{algorithmic}[1]
\REQUIRE $K$ clients, Global rounds $T$, DP budget $\varepsilon$
\STATE Server initializes global prototype bank $\mathcal{P}_{global}^{(0)} = \emptyset$
\FOR{$t = 1, \dots, T$}
    \FOR{each client $k \in \{1, \dots, K\}$ \textbf{in parallel}}
        \STATE Retrieve $\mathcal{P}_{global}^{(t-1)}$ from server
        \STATE Extract abstracted graphs $\tilde{\mathcal{G}}_k = \text{VCSA}(\mathcal{G}_k)$
        \STATE Update local GNN $\theta_k$ using MGMP with $\mathcal{P}_{global}^{(t-1)}$ for $E$ epochs
        \STATE Compute local prototypes $\mathbf{p}_c^{(k)} \leftarrow \frac{1}{N_{c,k}} \sum \text{GNN}_{\theta_k}(\tilde{G})$
        \STATE Perturb: $\hat{\mathbf{p}}_c^{(k)} \leftarrow \mathbf{p}_c^{(k)} + \text{Lap}(0, \Delta f / \varepsilon)$
        \STATE Transmit $\hat{\mathbf{p}}_c^{(k)}$ and $w_c^{(k)}$ to server
    \ENDFOR
    \STATE Server aggregates: $\mathbf{p}_c^{global, (t)} \leftarrow \sum_k w_c^{(k)} \hat{\mathbf{p}}_c^{(k)}$
\ENDFOR
\end{algorithmic}
\end{algorithm}
